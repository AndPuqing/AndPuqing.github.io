第二章

--创建数据库
CREATE DATABASE database_name 
ON
(
    NAME = logical_file_name,
    FILENAME
    ...
)
LOG ON
(
     NAME = logical_file_name,
    FILENAME
    ...
)

--修改数据课信息
ALTER DATABASE database_name
{
	ADD FILE<filespec> --添加数据文件
	MODIFY FILE <filespec> --修改数据文件
	MODIFY NAME = new_dbname --修改数据库名
}

--手动收缩数据库
DBSS SHRINKDATABASE (database_name ,[target_percent])

--修改数据库名
sp_renamedb 原,新

--删除数据库
DROP DATABASE dbname

--查看数据库信息
sp_helpdb[数据库名]

第三章

--创建表
CREATE TABLE<表名>(
	<列名><数据类型> [NULL| NOT NULL] [IDENTITY[(seed,increment)]] [{<约束>}]
)

--修改表
ALTER TABLE<表名>(
ADD<列定义> --添加字段
DROP COLUMN<列名> --删除字段
ALTER COLUMN<列名><属性> --修改列属性
)

--修改列名
Sp_rename 'Student.Sdept','系别','column'

以下用实列代替标准写法

约束

在SQL Server中有五种约束：

  1. 主键约束（Primary Key）
  2. 唯一性约束（Unique）
  3. 检查约束（Check）
  4. 默认值约束（Default）
  5. 外键约束（Foreign Key）

而根据声明的形式又可分为列级约束,表级约束.

关于二者的区别，直接用图吧。

由上图所知主键、外键、唯一、检查这四项，既可以创建列约束，也可以创建表约束。而缺省和非空只能创建列约束。

约束创建

  1. 主键约束

    --在创建时创建
CREATE TABLE Course(
	Con char(6) NOT NULL PRIMARY KEY,
)

--使用表级创建
CREATE TABLE Course(
	Con char(6) NOT NULL,
    City varchar(255),
    CONSTRAINT pk_Sc PRIMARY KEY(Con),
    CONSTRAINT pk_CC PRIMARY KEY(Con,City)
)

--在现有表创建
ALTER TABLE SC
ADD CONSTRAINT PK_S PRIMARY KEY(Con,City)
