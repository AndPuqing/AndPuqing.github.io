约束

在SQL Server中有五种约束：

  1. 主键约束（Primary Key）
  2. 唯一性约束（Unique）
  3. 检查约束（Check）
  4. 默认值约束（Default）
  5. 外键约束（Foreign Key）

而根据声明的形式又可分为列级约束,表级约束.

关于二者的区别，直接用图吧。

由上图所知主键、外键、唯一、检查这四项，既可以创建列约束，也可以创建表约束。而缺省和非空只能创建列约束。

约束创建

  1. 主键约束
--在创建时创建
CREATE TABLE Course(
    Con char(6) NOT NULL PRIMARY KEY,
)

--使用表级创建
CREATE TABLE Course(
    Con char(6) NOT NULL,
    City varchar(255),
    CONSTRAINT pk_Sc PRIMARY KEY(Con),
    CONSTRAINT pk_CC PRIMARY KEY(Con,City)
)

--在现有表创建
ALTER TABLE SC
ADD CONSTRAINT PK_S PRIMARY KEY(Con,City)

--删除约束
DROP CONSTRAINT PK_S

似乎在SQL Server 2012中可以允许不指定约束名，而直接创建，但为了规范还是指定约束名。

  1. 外键约束
--在创建时创建
CREATE TABLE SC(
    Son int FOREIGN KEY REFERENCES student(Sno)
)

--使用表级创建
CREATE TABLE SC(
    Son int NOT NULL,
    City varchar(255),
    CONSTRAINT fk_p FOREIGN KEY (Sno)
    REFERENCES student(Sno)
)

--在现有表创建

--不指定名
ALTER TABLE SC
ADD FOREIGN KEY (Con) REFERENCES student(Cno)
--指定名
ALTER TABLE SC
ADD CONSTRAINT fk_per FOREIGN KEY (Con) REFERENCES student(Cno)
  1. 唯一约束
  • 唯一约束与主键约束的区别在于可以为NULL
  • 可以为多个
--在创建时创建
CREATE TABLE SC(
    Son int UNIQUE
)

--使用表级创建
CREATE TABLE SC(
    Son int NOT NULL,
    City varchar(255),
    CONSTRAINT uc_p UNIQUE (Son,City)
)

--在现有表创建

--不指定名
ALTER TABLE SC
ADD UNIQUE (Sno)
--指定名
ALTER TABLE SC
ADD CONSTRAINT uc_p UNIQUE (Cno)
  1. 检测约束
--在创建时创建
CREATE TABLE SC(
    Son int CHECK(Sno>0)
)

--使用表级创建
CREATE TABLE SC(
    Son int NOT NULL,
    City varchar(255),
    CONSTRAINT uc_p UNIQUE (Son>0 and City='changde')
)

--在现有表创建

--不指定名
ALTER TABLE SC
ADD CHECK (Sno>0)
--指定名
ALTER TABLE SC
ADD CONSTRAINT uc_p CHECK (Cno>0)
  1. 默认值约束
CREATE TABLE Persons
(
	Id_P int NOT NULL,
	City varchar(255) DEFAULT 'Sandnes'
)
--还可以这样玩
CREATE TABLE Orders
(
	OrderDate date DEFAULT GETDATE()
)

--ALTER
ALTER TABLE Persons
ADD DEFAULT 'llala' FOR City

小小总结

--创建时
Cno int PRIMARY KEY
Cno int FOREIGN KEY REFERENCES Student(Cno)
Cno int UNIQUE
Cno int CHECK(Cno>0)
Cno int DEFAULT '607'

--表级约束
CONSTRAINT <约束名> PRIMARY KEY (COLUMN,)
CONSTRAINT <约束名> FOREIGN KEY Sno REFERENCES student(Sno)
CONSTRAINT <约束名> UNIQUE (Sno)
CONSTRAINT <约束名> CHECK (Cno > 0 and City = 'la')
CONSTRAINT <约束名> DEFAULT '男' FOR Sex
--以上CONSTRAINT和约束名均可省略

--事后型
ALTER TABLE <表名>
ADD CONSTRAINT <约束名> PRIMARY KEY(Con,City)
ADD CONSTRAINT <约束名> FOREIGN KEY Sno REFERENCES Student(Sno)
ADD CONSTRAINT <约束名> UNIQUE(Con)
ADD CONSTRAINT <约束名> CHECK (Cno > 0 and City = 'la')
ADD CONSTRAINT <约束名> DEFAULT '男' FOR Sex

相关子查询

对于嵌套查询中的单值、多值查询是容易理解的，但是对于相关子查询是不容易理解的。

关键在于查询的流程不一样了。

--给定SQL语句
select * from student 
where []
group by []
having []
order by []

$$ \begin{aligned} &processing\ like\ this.\cr &From \to Where \to Group\ by\to Having\to Select\to Order\ By \end{aligned} $$

--但是对于一个相关子查询就不一样了
select * from student as s1
where sage>(
	select avg(sage) from student as s2
	where s1.sdept=s2.sdept
)

$$ \begin{aligned} &It’s\ different\ now. \cr &step\ 1. select\ one\ from\ student\ as\ s1\cr &step\ 2. from\ student\ as\ s2\cr &step\ 3. where\ s1.sdept=s2.sdept\cr &step\ 4. select\ AVG()\cr &step\ 5. where\ sage>AVG()\cr &step\ 6. ture\ next \cr &step\ 7. back\ step\ 1 \end{aligned} $$

所以相关子查询先从外查询取一条数据，再从子查询返回满足条件的数据，然后利用该数据再次查询。

子查询还有一个相关函数

[NOT] EXISTS(子查询)
SELECT * FROM student
where exists(
	select * from sc
    where cno=2 and sno=student.sno
)
--在这里exists判断了子查询是否为空，若为空，不返回值，若不为空，返回select *

CASE第一种写法

--CASE表达式
CASE 输入表达式
	WHEN 表达式1 THEN 结果1
	WHEN 表达式2 THEN 结果2
	WHEN 表达式3 THEN 结果3
	ELSE 其他结果
END

这种CASE的逻辑是，判断输入表达式是否满足表达式，若满足则退出，若都不满足则去ELSE

第二种

CASE
	WHEN 逻辑表达式1 THEN 结果表达式1
	WHEN 逻辑表达式2 THEN 结果表达式2
	WHEN 逻辑表达式3 THEN 结果表达式3
	[...n]
	ELSE 其他结果表达式
END

二者的区别在于第一种只能对一个表达式判断，而后者则能对不同对象进行逻辑判断。
